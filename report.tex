	\section*{1. Δομές Πίνακα Συμβόλων (\en{Symbol Table Classes})}
	
	\textbf{\en{class\_method}:}
	\begin{itemize}[noitemsep,topsep=0pt,leftmargin=*]
		\item \texttt{\en{\_\_init\_\_}}: Αρχικοποίηση (\en{name}, \en{constructor flag}, πεδία, παράμετροι, \en{version}).
		\item \texttt{\en{set\_version / get\_version}}: Διαχείριση εκδόσεων για υπερφόρτωση (\en{overloading}).
		\item \texttt{\en{set\_return\_type / get\_return\_type}}: Διαχείριση τύπου επιστροφής.
		\item \texttt{\en{add\_field}}: Προσθήκη τοπικών μεταβλητών ή παραμέτρων.
		\item \texttt{\en{has\_field / get\_field\_type}}: Έλεγχος ύπαρξης και ανάκτηση τύπου μεταβλητής.
		\item \texttt{\en{get\_method\_with\_version}}: Επιστρέφει το μοναδικό όνομα (\texttt{\en{name\_version}}).
	\end{itemize}
	
	\textbf{\en{class\_info}:}
	\begin{itemize}[noitemsep,topsep=0pt,leftmargin=*]
		\item \texttt{\en{add\_inheritance}}: Προσθήκη γονικής κλάσης.
		\item \texttt{\en{add\_method}}: Δημιουργία μεθόδου με  διαχείριση \en{versioning}.
		\item \texttt{\en{add\_field / has\_field}}: Διαχείριση πεδίων κλάσης (με έλεγχο κληρονομικότητας).
		\item \texttt{\en{search\_method}}: Αναζήτηση μεθόδου βάσει ονόματος και αριθμού παραμέτρων.
		\item \texttt{\en{check\_if\_overide\_methods\_valid}}: Έλεγχος εγκυρότητας \en{overloading}.
	\end{itemize}
	
	\section*{2. Κλάση \en{Listener} (\en{oosMyListener})}
	
	\begin{multicols}{2}
		
		\subsection*{Βοηθητικές Μέθοδοι}
		\begin{itemize}[leftmargin=*,noitemsep]
			\item \texttt{\en{add\_class}}: Εγγραφή νέας κλάσης.
			\item \texttt{\en{add\_field\_to\_class}}: Προσθήκη πεδίου σε \en{struct}.
			\item \texttt{\en{add\_method\_to\_class}}: Εγγραφή νέας μεθόδου.
			\item \texttt{\en{get\_class\_obj}}: Ανάκτηση πληροφοριών κλάσης.
			\item \texttt{\en{chech\_if\_id\_declared}}: Ελέγχει αν μια μεταβλητή έχει δηλωθεί (είτε ως πεδίο της κλάσης είτε ως τοπική μεταβλητή).
			\item \texttt{\en{get\_oos\_compiled}}: Εγγραφή του κώδικα \en{C} στο αρχείο εξόδου.
			\item \texttt{\en{tabbing / indent / deindent}}: Μορφοποίηση κώδικα \en{C}.
		\end{itemize}
		
		\subsection*{Δηλώσεις (\en{Definitions})}
		\begin{itemize}[leftmargin=*,noitemsep]
			\item \texttt{\en{enterStartRule}}: Προσθήκη \en{headers} (\texttt{\en{stdio.h}}, \texttt{\en{stdlib.h}}).
			\item \texttt{\en{enterClass\_def}}: Δημιουργία \texttt{\en{struct}}, χειρισμός κληρονομικότητας
			\item \texttt{\en{enterClass\_body}}: Κλείσιμο του \en{struct}.
			\item \texttt{\en{exitDecl\_line}}: Δημιουργία δηλώσεων πεδίων (\texttt{\en{int}} ή \texttt{\en{struct*}}).
			\item \texttt{\en{enterConstructor\_def}}: Ορισμός συνάρτησης \texttt{\en{init}}
			\item \texttt{\en{enterMethod\_def}}: Ορισμός \en{C} συνάρτησης (\texttt{\en{Class\_Method\_Ver}}).
			\item \texttt{\en{enterClass\_main\_def}}: Δημιουργία της \texttt{\en{main()}} με \en{print} έναρξης.
		\end{itemize}
		
		\subsection*{Διαχείριση Παραμέτρων}
		\begin{itemize}[leftmargin=*,noitemsep]
			\item \texttt{\en{exitParlist}}: Παραγωγή ορισμάτων 
		\end{itemize}
		
		\subsection*{Εντολές (\en{Statements})}
		\begin{itemize}[leftmargin=*,noitemsep]
			\item \texttt{\en{enterReturn\_stat}}: Χειρισμός \texttt{\en{return self\_ptr}} ή τιμών.
			\item \texttt{\en{enterAssignment\_stat}}: Ανάθεση τιμών (\texttt{\en{self\_ptr->field = ...}}).
			\item \texttt{\en{enterDirect\_call\_stat}}: Κλήσεις \en{void} μεθόδων.
			\item \texttt{\en{enterPrint\_stat}}: Μετατροπή σε \texttt{\en{printf}}.
			\item \texttt{\en{enterInput\_stat}}: Μετατροπή σε \texttt{\en{scanf}}.
			\item \texttt{\en{enterIf/While\_stat}}: Δομές ελέγχου .
		\end{itemize}
		
		\subsection*{Εκφράσεις (\en{Expressions})}
		\begin{itemize}[leftmargin=*,noitemsep]
			\item \texttt{\en{enterFactor}}: Ανάλυση τελεστών. Χειρισμός:
			\begin{itemize}[nosep]
				\item Ακέραιοι (\texttt{\en{INTEGER}})
				\item Πεδία (\texttt{\en{self.id}} $\to$ \texttt{\en{self\_ptr->id}})
				\item Κλήσεις μεθόδων (\texttt{\en{self.obj.func()}})
				\item \en{Nested access} (\texttt{\en{obj.field}})
			\end{itemize}
			\item \texttt{\en{enterAdd/Mul\_oper}}: Προσθήκη τελεστών στο \en{stack}.
		\end{itemize}
		
		\subsection*{Κλήσεις Συναρτήσεων}
		\begin{itemize}[leftmargin=*,noitemsep]
			\item \texttt{\en{enterFunc\_call}}: Προετοιμασία \en{stack} κλήσης.
			\item \texttt{\en{exitFunc\_call}}: 
			\begin{itemize}[nosep]
				\item Εύρεση σωστής έκδοσης (\en{version})
				\item \en{Casting} για κληρονομικότητα (\texttt{\en{\&obj->parent}})
				\item Διαχείριση \texttt{\en{NULL}} για \en{constructors}
				\item Παραγωγή τελικού \en{C string}
			\end{itemize}
		\end{itemize}
		
	\end{multicols}
